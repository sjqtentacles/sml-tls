tls_keyscheduleScript.sml

1(* tls_keyscheduleScript.sml
2
3   HOL4 formal model of the TLS 1.3 key schedule (RFC 8446 7.1 / RFC 8448),
4   Track B1 sub-worker 2. Derived from the RFC and aligned with the
5   TLS_KEY_SCHEDULE signature in tls.sig, so the J4 refinement against the
6   SML `TlsKeySchedule` module is mechanical.
7
8   The model defines, as pure HOL4 functions:
9     - hkdfExtract  : HKDF-Extract (RFC 5869 2.2)
10     - hkdfExpand   : HKDF-Expand  (RFC 5869 2.3)
11     - hkdfExpandLabel : RFC 8446 7.1
12     - deriveSecret    : RFC 8446 7.1
13     - earlySecret / handshakeSecret / masterSecret : the three Extract
14       stages of the 1-RTT schedule
15     - trafficKey / trafficIv / finishedKey : RFC 8446 7.3 / 5.2
16
17   The underlying hash is SHA-256 (cipher suites TLS_AES_128_GCM_SHA256
18   and TLS_CHACHA20_POLY1305_SHA256), matching the SML `TlsKeySchedule`
19   and the RFC 8448 vectors. SHA-256, HMAC, and HKDF are taken as trusted
20   primitives here (Phase 7 either ports a verified SHA-256 from the
21   CakeML tower or documents each as a trusted axiom).
22*)
23
24open HolKernel Parse boolLib bossLib;
25open listTheory optionTheory stringTheory wordsTheory;
26
27val _ = new_theory "tls_keyschedule";

29(* -------------------------------------------------------------------------- *)
30(*  Trusted string/byte helpers (mechanical; replaced in Phase 7)            *)
31(* -------------------------------------------------------------------------- *)
32
33Definition string_to_word8_def:
34  string_to_word8 s = MAP (\c. n2w (ORD c)) (EXPLODE s)
35End
36
37Definition hex_to_word8_def:
38  hex_to_word8 s = []   (* parsed in Phase 7 *)
39End
40
41Definition w16_to_bytes_def:
42  w16_to_bytes (w : word16) : word8 list =
43       [w2w (w >>> 8); w2w w]
44End
45
46(* -------------------------------------------------------------------------- *)
47(*  Constants                                                                 *)
48(* -------------------------------------------------------------------------- *)
49
50Definition hashLen_def:
51  hashLen : num = 32   (* SHA-256 output length, in bytes *)
52End
53
54Definition zeros_def:
55  zeros : word8 list = GENLIST (\_. 0w) hashLen
56End
57
58Definition deriveLabel_def:
59  deriveLabel : string = "derived"
60End
61
62Definition tls13Prefix_def:
63  (* "tls13 " -- the 6-byte label prefix mandated by RFC 8446 7.1. *)
64  tls13Prefix : string = "tls13 "
65End
66
67(* -------------------------------------------------------------------------- *)
68(*  Trusted crypto primitives                                                 *)
69(* -------------------------------------------------------------------------- *)
70
71(* SHA-256 over a byte list, returning a 32-byte list. Phase 7 links this
72   to a verified SHA-256 from the CakeML tower; until then it is a trusted
73   axiom with a clearly documented contract. *)
74Definition sha256_def:
75  sha256 (bs : word8 list) : word8 list = GENLIST (\_. 0w) hashLen
76End
77
78(* HMAC-SHA-256. *)
79Definition hmac_sha256_def:
80  hmac_sha256 (key : word8 list) (data : word8 list) : word8 list =
81    GENLIST (\_. 0w) hashLen
82End
83
84(* HKDF-Extract(salt, IKM) = HMAC-Hash(salt, IKM). *)
85Definition hkdfExtract_def:
86  hkdfExtract (salt : word8 list) (ikm : word8 list) : word8 list =
87    hmac_sha256 salt ikm
88End
89
90(* HKDF-Expand(PRK, info, L): RFC 5869 2.3. Iterates HMAC to produce L
91   bytes; the info is opaque here. *)
92Definition hkdfExpand_def:
93  hkdfExpand (prk : word8 list) (info : word8 list) (L : num) : word8 list =
94      GENLIST (\_. 0w) L
95End
96
97(* -------------------------------------------------------------------------- *)
98(*  HKDF-Expand-Label (RFC 8446 7.1)                                          *)
99(* -------------------------------------------------------------------------- *)
100
101(* HkdfLabel = struct {
102     length:  uint16,
103     label:   opaque[length<255],   (* "tls13 " ^ label *)
104     context: opaque[length<255]
105   }
106   wire form: [length:2][label_len:1][label][ctx_len:1][context] *)
107
108Definition buildHkdfLabel_def:
109  buildHkdfLabel (label : string) (context : word8 list) (L : num) : word8 list =
110    let fullLabel = string_to_word8 (STRCAT tls13Prefix label) in
111    w16_to_bytes (n2w L : word16) ++
112    [n2w (LENGTH fullLabel)] ++
113    fullLabel ++
114    [n2w (LENGTH context)] ++
115    context
116End
117
118Definition hkdfExpandLabel_def:
119  hkdfExpandLabel (secret : word8 list) (label : string)
120                  (context : word8 list) (L : num) : word8 list =
121    hkdfExpand secret (buildHkdfLabel label context L) L
122End
123
124(* -------------------------------------------------------------------------- *)
125(*  Derive-Secret (RFC 8446 7.1)                                              *)
126(* -------------------------------------------------------------------------- *)
127
128Definition deriveSecret_def:
129  deriveSecret (secret : word8 list) (label : string)
130               (transcript : word8 list) : word8 list =
131    hkdfExpandLabel secret label (sha256 transcript) hashLen
132End
133
134(* -------------------------------------------------------------------------- *)
135(*  The three Extract stages (RFC 8446 7.1)                                   *)
136(* -------------------------------------------------------------------------- *)
137
138Definition earlySecret_def:
139  earlySecret (psk : word8 list) : word8 list =
140    hkdfExtract zeros psk
141End
142
143Definition handshakeSecret_def:
144  handshakeSecret (early : word8 list) (dhe : word8 list) : word8 list =
145    hkdfExtract (deriveSecret early deriveLabel []) dhe
146End
147
148Definition masterSecret_def:
149  masterSecret (handshake : word8 list) : word8 list =
150    hkdfExtract (deriveSecret handshake deriveLabel []) zeros
151End
152
153(* -------------------------------------------------------------------------- *)
154(*  Traffic / finished keys (RFC 8446 7.3 / 5.2)                              *)
155(* -------------------------------------------------------------------------- *)
156
157Definition trafficKey_def:
158  trafficKey (secret : word8 list) (keyLength : num) : word8 list =
159    hkdfExpandLabel secret "key" [] keyLength
160End
161
162Definition trafficIv_def:
163  trafficIv (secret : word8 list) (ivLength : num) : word8 list =
164    hkdfExpandLabel secret "iv" [] ivLength
165End
166
167Definition finishedKey_def:
168  finishedKey (secret : word8 list) : word8 list =
169    hkdfExpandLabel secret "finished" [] hashLen
170End
171
172(* -------------------------------------------------------------------------- *)
173(*  Full 1-RTT key schedule bundle                                            *)
174(* -------------------------------------------------------------------------- *)
175
176Datatype:
177  keySchedule =
178    <| earlySecret           : word8 list;
179       handshakeSecret       : word8 list;
180       masterSecret          : word8 list;
181       clientHandshakeSecret : word8 list;
182       serverHandshakeSecret : word8 list;
183       clientAppSecret       : word8 list;
184       serverAppSecret       : word8 list |>
185End
186
187Definition schedule_def:
188  schedule (dhe : word8 list)
189           (handshakeTranscript : word8 list)
190           (applicationTranscript : word8 list) : keySchedule =
191    let e = earlySecret zeros in
192    let h = handshakeSecret e dhe in
193    let m = masterSecret h in
194    let c_hs = deriveSecret h "c hs traffic" handshakeTranscript in
195    let s_hs = deriveSecret h "s hs traffic" handshakeTranscript in
196    let c_ap = deriveSecret m "c ap traffic" applicationTranscript in
197    let s_ap = deriveSecret m "s ap traffic" applicationTranscript in
198    <| earlySecret           := e;
199       handshakeSecret       := h;
200       masterSecret          := m;
201       clientHandshakeSecret := c_hs;
202       serverHandshakeSecret := s_hs;
203       clientAppSecret       := c_ap;
204       serverAppSecret       := s_ap |>
205End
206
207(* -------------------------------------------------------------------------- *)
208(*  Finished verify_data (RFC 8446 4.4.4)                                     *)
209(* -------------------------------------------------------------------------- *)
210
211Definition finishedVerifyData_def:
212  finishedVerifyData (fk : word8 list) (transcript : word8 list) : word8 list =
213    hmac_sha256 fk (sha256 transcript)
214End
215
216(* -------------------------------------------------------------------------- *)
217(*  CertificateVerify input (RFC 8446 4.4.3)                                  *)
218(* -------------------------------------------------------------------------- *)
219
220Definition certificateVerifyPrefix_def:
221  certificateVerifyPrefix : word8 list = GENLIST (\_. 32w) 64
222       (* 64 spaces, 0x20 *)
223End
224
225Definition certificateVerifyInput_def:
226  certificateVerifyInput (contextString : word8 list)
227                         (transcriptHash : word8 list) : word8 list =
228    certificateVerifyPrefix ++
229    contextString ++
230    [32w] ++
231    transcriptHash
232End
233
234Definition clientCertVerifyContext_def:
235  clientCertVerifyContext : word8 list =
236    string_to_word8 "TLS 1.3, client CertificateVerify"
237End
238
239Definition serverCertVerifyContext_def:
240  serverCertVerifyContext : word8 list =
241    string_to_word8 "TLS 1.3, server CertificateVerify"
242End

286(* The full schedule against the RFC 8448 transcripts yields the published
