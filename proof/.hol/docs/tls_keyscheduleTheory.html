schedule_def
⊢ ∀dhe handshakeTranscript applicationTranscript.
    schedule dhe handshakeTranscript applicationTranscript =
    (let
       e = earlySecret zeros;
       h = handshakeSecret e dhe;
       m = masterSecret h;
       c_hs = deriveSecret h "c hs traffic" handshakeTranscript;
       s_hs = deriveSecret h "s hs traffic" handshakeTranscript;
       c_ap = deriveSecret m "c ap traffic" applicationTranscript;
       s_ap = deriveSecret m "s ap traffic" applicationTranscript
     in
       <|earlySecret := e; handshakeSecret := h; masterSecret := m;
         clientHandshakeSecret := c_hs; serverHandshakeSecret := s_hs;
         clientAppSecret := c_ap; serverAppSecret := s_ap|>)
