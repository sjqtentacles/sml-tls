1(* tls_wireScript.sml
2
3   HOL4 formal model of the TLS 1.3 wire formats (RFC 8446), Track B1,
4   sub-worker 1. This theory is derived from the RFC (and aligned with
5   the Phase 0 SML signatures TLS_RECORD / TLS_HANDSHAKE / TLS_EXTENSIONS
6   in tls.sig), so it proceeds in parallel with all functional work.
7
8   Conventions mirror the SML library:
9     - bytes are HOL4 ``:word8`` values; byte strings are ``:word8 list``;
10     - multi-byte integers are big-endian;
11     - every decoder is total: a malformed input yields ``NONE``, never a
12       partial value;
13     - datatypes are exhaustive over the RFC's named values and carry an
14       ``Other`` constructor only where the SML does (alert descriptions).
15
16   Round-trip theorems of the form
17
18       decode (encode x) = SOME (x, [])
19
20   are stated (and proved by EVAL for the small cases) for each structure.
21*)
22

37(* High/low byte extraction for the 16-bit big-endian encoder. *)
38Definition w8_of_w16_hi_def:
39  w8_of_w16_hi (w : word16) : word8 = w2w (w >>> 8)
40End
41
42Definition w8_of_w16_lo_def:
43  w8_of_w16_lo (w : word16) : word8 = w2w w
44End
45
46(* w16_of_bytes hi lo = (hi << 8) | lo, big-endian reassembly. *)
47Definition w16_of_bytes_def:
48  w16_of_bytes (hi : word8) (lo : word8) : word16 =
49    (w2w hi << 8) || w2w lo
50End
51
52(* w24_of_len3 hi mid lo reassembles a 24-bit length. *)
53Definition w24_of_len3_def:
54  w24_of_len3 (hi : word8) (mid : word8) (lo : word8) : word32 =
55    (w2w hi << 16) || (w2w mid << 8) || w2w lo
56End
57
58(* len3 n splits a 24-bit length into three bytes, high byte first. *)
59Definition len3_def:
60  len3 (n : num) : word8 list =
61    [n2w ((n DIV 65536) MOD 256);
62     n2w ((n DIV 256) MOD 256);
63     n2w (n MOD 256)]
64End

74(* The TLS legacy record version field: always 0x0303 on the wire even for
75   TLS 1.3 (RFC 8446 5.1). Kept as a 16-bit word constant. *)
76Definition legacyVersion_def:
77  legacyVersion : word16 = 0x0303w
78End
79
80(* Encode a 16-bit big-endian integer as a 2-byte list. The high byte is
81   the top 8 bits; the low byte is the bottom 8. Defined in terms of the
82   w8_of_w16_hi/lo helpers axiomatised at the bottom of this file. *)
83Definition w16_to_bytes_def:
84  w16_to_bytes (w : word16) : word8 list =
85    [w8_of_w16_hi w; w8_of_w16_lo w]
86End

92Datatype:
93  contentType =
94      Invalid
95    | ChangeCipherSpec
96    | Alert
97    | Handshake
98    | ApplicationData
99End
100
101Definition encodeContentType_def:
102  (encodeContentType Invalid          = 0x00w : word8) /\
103  (encodeContentType ChangeCipherSpec = 0x14w) /\
104  (encodeContentType Alert            = 0x15w) /\
105  (encodeContentType Handshake        = 0x16w) /\
106  (encodeContentType ApplicationData  = 0x17w)
107End
108
109Definition decodeContentType_def:
110  (decodeContentType (b : word8) =
111     if b = 0x00w then SOME Invalid else
112     if b = 0x14w then SOME ChangeCipherSpec else
113     if b = 0x15w then SOME Alert else
114     if b = 0x16w then SOME Handshake else
115     if b = 0x17w then SOME ApplicationData else
116     NONE)
117End

119Theorem decode_encode_contentType:
120  !ct. decodeContentType (encodeContentType ct) = SOME ct
121Proof
122  cheat
123QED

129Datatype:
130  tlsPlaintext =
131    <| contentType : contentType;
132       (* legacy_record_version is always legacyVersion, so it is omitted
133          from the model and re-introduced on encode. *)
134       fragment     : word8 list |>
135End
136
137Datatype:
138  tlsCiphertext =
139    <| contentType       : contentType;
140       encryptedRecord   : word8 list |>
141End
142
143(* 5-byte header: [type:1][version:2][length:2], then the fragment.
144   Length is the fragment length; we assume < 2^14 (enforced elsewhere). *)
145Definition encodePlaintext_def:
146  encodePlaintext (r : tlsPlaintext) : word8 list =
147    encodeContentType r.contentType ::
148    (w16_to_bytes legacyVersion ++
149     w16_to_bytes (n2w (LENGTH r.fragment) : word16) ++
150     r.fragment)
151End
152
153Definition decodePlaintext_def:
154  decodePlaintext (bs : word8 list) : (tlsPlaintext # word8 list) option =
155    if LENGTH bs < 5 then NONE
156    else
157      let ctB   = EL 0 bs in
158      let n     = w2n (w16_of_bytes (EL 3 bs) (EL 4 bs)) in
159      case decodeContentType ctB of
160        | NONE => NONE
161        | SOME ct =>
162            if LENGTH bs < 5 + n then NONE
163            else SOME (<| contentType := ct; fragment := TAKE n (DROP 5 bs) |>,
164                       DROP (5 + n) bs)
165End
166
167(* Round-trip: decoding an encoded record returns the record and no
168   remainder. Provable by EVAL on concrete records; the general case
169   reduces to list arithmetic. *)
170Theorem decode_encode_plaintext:
171  !r. decodePlaintext (encodePlaintext r) = SOME (r, [])
172Proof
173  (* TODO: discharge in Phase 7; the case-splits are mechanical. *)
174  cheat
175QED

177(* Ciphertext framing is identical in shape (outer type is always
178   ApplicationData under a traffic key, but the wire format is the same). *)

181    encodeContentType r.contentType ::
182    (w16_to_bytes legacyVersion ++
183     w16_to_bytes (n2w (LENGTH r.encryptedRecord) : word16) ++
184     r.encryptedRecord)
185End
186

262(* -------------------------------------------------------------------------- *)
263(*  Handshake message framing (RFC 8446 4)                                    *)
264(* -------------------------------------------------------------------------- *)
265
266Datatype:
267  handshakeMessage =
268    <| msgType : handshakeType;
269       body    : word8 list |>
270End
271
272(* A handshake message on the wire is [type:1][length:3][body]. The length
273   is a 24-bit big-endian integer; we model it as three word8 bytes. *)
274Definition encodeMessage_def:
275  encodeMessage (m : handshakeMessage) : word8 list =
276    encodeHandshakeType m.msgType ::
277    (len3 (LENGTH m.body) ++ m.body)
278End
279
280Definition decodeMessage_def:
281  decodeMessage (bs : word8 list) : (handshakeMessage # word8 list) option =
282    if LENGTH bs < 4 then NONE
283    else
284      let t = EL 0 bs in
285      let n = w2n (w24_of_len3 (EL 1 bs) (EL 2 bs) (EL 3 bs)) in
286      case decodeHandshakeType t of
287        | NONE => NONE
288        | SOME ht =>
289            if LENGTH bs < 4 + n then NONE
290            else SOME (<| msgType := ht; body := TAKE n (DROP 4 bs) |>,
291                       DROP (4 + n) bs)
292End
293
294Theorem decode_encode_message:
295  !m. decodeMessage (encodeMessage m) = SOME (m, [])
296Proof
297  cheat
298QED

300(* -------------------------------------------------------------------------- *)
301(*  Extensions (RFC 8446 4.2)                                                *)
302(* -------------------------------------------------------------------------- *)
303
304Datatype:
305  extension = <| extType : word16; data : word8 list |>
306End
307
308(* Extension block = 2-byte total length, then a sequence of entries,
309   each [type:2][length:2][data]. *)
310Definition encodeExtension_def:
311  encodeExtension (e : extension) : word8 list =
312    w16_to_bytes e.extType ++
313    w16_to_bytes (n2w (LENGTH e.data) : word16) ++
314    e.data
315End
316
317Definition encodeExtensions_def:
318  encodeExtensions (es : extension list) : word8 list =
319    let body = FLAT (MAP encodeExtension es) in
320    w16_to_bytes (n2w (LENGTH body) : word16) ++ body
321End
322
323(* Tail-recursive parser over the extension body. Each iteration reads a
324   4-byte entry header; a truncated length, a length exceeding the
325   remaining budget, or trailing junk after `total` bytes yields NONE. *)
326Definition decodeExts_loop_def:
327  (decodeExts_loop [] acc = SOME (REVERSE acc)) /\
328  (decodeExts_loop (h0::h1::l0::l1::rest) acc =
329     let extType = w16_of_bytes h0 h1 in
330     let n       = w2n (w16_of_bytes l0 l1) in
331     if LENGTH rest < n then NONE
332     else decodeExts_loop (DROP n rest)
333                          (<| extType := extType; data := TAKE n rest |> :: acc)) /\
334  (decodeExts_loop [_] acc = NONE) /\
335  (decodeExts_loop [_;_] acc = NONE) /\
336  (decodeExts_loop [_;_;_] acc = NONE)
337Termination
338  WF_REL_TAC `measure (LENGTH o FST)` \\ rw [] \\ DECIDE_TAC
339End
340
341Definition decodeExtensions_def:
342  decodeExtensions (bs : word8 list) : extension list option =
343    if LENGTH bs < 2 then NONE
344    else
345      let total = w2n (w16_of_bytes (EL 0 bs) (EL 1 bs)) in
346      if LENGTH bs < 2 + total then NONE
347      else decodeExts_loop (TAKE total (DROP 2 bs)) []
348End
349
350Theorem decode_encode_extensions:
351  !es. decodeExtensions (encodeExtensions es) = SOME es
352Proof
353  cheat
354QED
